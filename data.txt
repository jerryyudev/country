<body>
  <div class="container">
    <h2>🌐 全球国家风险评级查询</h2>
    <label for="country">请输入国家英文名称或拼音（如 china / afghanistan）：</label>
    <input type="text" id="country" oninput="handleInput(this.value)" placeholder="请输入国家名称...">
    <div class="suggestions" id="suggestions">🔄 加载中，请稍候...</div>
    <div class="result" id="result"></div>
  </div>

  <script>
    let data = {};
    let allCountries = [];

    // 加载 data.json 文件
    fetch('./data.json')
      .then(res => res.json())
      .then(json => {
        data = {};
        allCountries = json.map(entry => entry.name);
        json.forEach(entry => {
          data[entry.name.toLowerCase()] = entry;
        });
        document.getElementById('suggestions').innerHTML = '✅ 数据加载完毕，请开始输入国家名称';
      })
      .catch(err => {
        document.getElementById('suggestions').innerHTML = '❌ 加载失败，请检查 data.json 是否存在';
      });

    const handleInput = (val) => {
      const result = document.getElementById('result');
      const sugg = document.getElementById('suggestions');
      result.innerHTML = '';
      if (!val.trim()) {
        sugg.innerHTML = '请输入国家名称，例如 china、iran、russia...';
        return;
      }

      const input = val.toLowerCase();
      const matches = allCountries.filter(c => c.toLowerCase().includes(input));
      if (matches.length > 0) {
        sugg.innerHTML = '建议：' + matches.map(match => `<span onclick="fillCountry('${match}')">${match}</span>`).join('、');
      } else {
        sugg.innerHTML = '未找到匹配项';
      }

      // 自动展示匹配内容
      const bestMatch = data[input] || data[matches[0]?.toLowerCase()];
      if (bestMatch) simulateStreaming(bestMatch);
    };

    const fillCountry = (name) => {
      document.getElementById('country').value = name;
      handleInput(name);
    };

    const simulateStreaming = (info) => {
      const r = document.getElementById('result');
      r.innerHTML = '';
      const lines = [
        `🌍 国家：${info.name}`,
        `🆔 国家代码：${info.code}`,
        `⚠️ 风险等级：${info.risk_level}/10`,
        `🔖 标签：${info.tags.join('、')}`,
        `📌 风险理由：${info.reason}`,
        `📚 知名案例：${info.notorious_cases}`,
        `🧳 旅游建议：${info.travel_advice}`,
        `💼 商业建议：${info.business_advice}`,
        `📝 总结：${info.summary}`
      ];

      let i = 0;
      const printLine = () => {
        if (i < lines.length) {
          r.innerHTML += lines[i] + '\n';
          i++;
          setTimeout(printLine, 120); // 流式延迟
        }
      };
      printLine();
    };
  </script>
